### 给你一棵二叉树的根节点 root ，返回其节点值的 后序遍历 。。

- 方法一：递归  
思路：左右后的顺序递归调用，注意，递归方法定义在入口方法内部 
```
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var postorderTraversal = function(root) {
    const result = []
    if(root===null){
        return result;
    }
    const order = (node)=>{
        node.left && order(node.left)
        node.right && order(node.right)
        result.push(node.val)
    }
    order(root)
    return result;
};
```

- 方法二：迭代  
思路：
  - 用数组模拟栈，先进后出，后序遍历：左右后  
  - 先沿着树左节点走到底，每一项放到栈中，走到底后判断其右子节点是否存在，不存在直接弹出，每次弹出时记录该弹出的值为prev
  - 若存在，判断其下的左子节点是否存在，存在继续上边操作，每次弹出时判断是否时prev，若是则无需弹出，不是则弹出，并更新prev

/**

 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var postorderTraversal = function(root) {
    const result = []
    if(root===null){
        return result;
    }
    
    const stack = []
    let temp = root
    while(temp!==null){
        stack.push(temp)
        temp = temp.left
    }
    let prev = null

    while(stack.length>0){
        const top = stack[stack.length -1]
        if(top.right === null){
            prev = stack.pop()
            result.push(top.val)
        }else{
            if(prev !== top.right){
                let temp2 = top.right
                while(temp2!==null){
                    stack.push(temp2)
                    temp2=temp2.left
                }
            } else {
                prev = stack.pop()
                result.push(top.val)
            }
        }
       
    }
    return result
};
