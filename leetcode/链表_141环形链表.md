var hasCycle = function(head) {
    if(!head||!head.next){
        return false;
    }
    let slowPointer = head;
    let fastPointer = head.next;
    while(slowPointer!==fastPointer){
        if(!fastPointer||!fastPointer.next){
            return false;
        }
        slowPointer = slowPointer.next;
        fastPointer = fastPointer.next.next;
    }
    return true
};
